# build runtime-lib in package 'runtime'
# update version in 'src/main/runtimeSE/package-info.java'

runtimelib="target/runtimeSE-1.0.2-RELEASE.jar"

rm -rf src/main/application \
    src/tests \
    src/resources

mk clean compile javadoc && \
    jar -c -v -f "$runtimelib" \
        -C target/classes . &&
        [ -f "$runtimelib" ] &&
          echo -e "-->\\ncreated: $runtimelib" ||
          echo -e "-->\\nno compiled classes or manifest, no .jar created"


# push javadoc to GitHub-pages
# 
tar cvf target/javadoc.tar target/javadoc
git restore .
git switch gh-pages
tar xvf target/javadoc.tar
git add target/javadoc && git commit -m "update, runtimeSE-1.0.2-RELEASE" && git push
git switch main
